@create Bulletin Board
@set Bulletin Board=QUIET
@lock/basic Bulletin Board==#1
@lock/enter Bulletin Board==#1

&DESCRIBE Bulletin Board=A large cube with a corkboard surface that residents and travelers can use to post notes.%rTo read the notes, type [ansi(h,read)]. To leave a note, type [ansi(h,write <message>)].
